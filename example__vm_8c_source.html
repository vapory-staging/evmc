79 static struct evmc_result execute(struct evmc_instance* instance,
80  struct evmc_context* context,
81  enum evmc_revision rev,
82  const struct evmc_message* msg,
83  const uint8_t* code,
84  size_t code_size)
85 {
86  struct evmc_result ret = {.status_code = EVMC_INTERNAL_ERROR};
87  if (code_size == 0)
88  {
89  // In case of empty code return a fancy error message.
90  const char* error = rev == EVMC_BYZANTIUM ? "Welcome to Byzantium!" : "Hello Ethereum!";
91  ret.output_data = (const uint8_t*)error;
92  ret.output_size = strlen(error);
93  ret.status_code = EVMC_FAILURE;
94  ret.release = NULL; // We don't need to release the constant messages.
95  return ret;
96  }

98  struct example_vm* vm = (struct example_vm*)instance;
99 
100  // Simulate executing by checking for some code patterns.
101  // Solidity inline assembly is used in the examples instead of EVM bytecode.
102 
103  // Assembly: `{ mstore(0, address()) return(0, msize()) }`.
104  const char return_address[] = "\x30\x60\x00\x52\x59\x60\x00\xf3";
105 
106  // Assembly: `{ sstore(0, add(sload(0), 1)) }`
107  const char counter[] = "\x60\x01\x60\x00\x54\x01\x60\x00\x55";
108 
109  if (code_size == strlen(return_address) &&
110  strncmp((const char*)code, return_address, code_size) == 0)
111  {
112  static const size_t address_size = sizeof(msg->destination);
113  uint8_t* output_data = (uint8_t*)malloc(address_size);
114  if (!output_data)
115  {
116  // malloc failed, report internal error.
117  ret.status_code = EVMC_INTERNAL_ERROR;
118  return ret;
119  }
120  memcpy(output_data, &msg->destination, address_size);
121  ret.status_code = EVMC_SUCCESS;
122  ret.output_data = output_data;
123  ret.output_size = address_size;
124  ret.release = &free_result_output_data;
125  return ret;
126  }
127  else if (code_size == strlen(counter) && strncmp((const char*)code, counter, code_size) == 0)
128  {
129  const evmc_bytes32 key = {{0}};
130  evmc_bytes32 value = context->host->get_storage(context, &msg->destination, &key);
131  value.bytes[31]++;
132  context->host->set_storage(context, &msg->destination, &key, &value);
133  ret.status_code = EVMC_SUCCESS;
134  return ret;
135  }
136 
137  ret.status_code = EVMC_FAILURE;
138  ret.gas_left = 0;
139 
140  if (vm->verbose)
141  printf("Execution done.\n");
142 
143  return ret;
144 }
